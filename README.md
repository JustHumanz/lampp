# install lampp

  Script for automatically install lampp(nginx,mariadb,phpmyadmin,php7.2) in ubuntu 18.04  
  
<p align="center">
  <b> How to install? </b>
</p>  

cd ~/  
git clone https://github.com/JustHumanz/lampp  
cd lampp  
chmod +x pack.sh  
sudo ./pack.sh
